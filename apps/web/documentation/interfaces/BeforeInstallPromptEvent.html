import { isPlatformBrowser } from '@angular/common';
import { Injectable, signal, inject, PLATFORM_ID } from '@angular/core';

interface BeforeInstallPromptEvent extends Event {
  readonly platforms: string[];
  prompt: () => Promise<void>;
  userChoice: Promise<{ outcome: 'accepted' | 'dismissed'; platform: string }>;
}

type NavigatorWithStandalone = Navigator & { standalone?: boolean };
type WindowWithMSStream = Window & { MSStream?: unknown };

/**
 * üì≤ PWA Install Service
 *
 * Maneja la instalaci√≥n de PWA con prompt personalizado
 */
@Injectable({
  providedIn: 'root',
})
export class PwaInstallService {
  private readonly platformId = inject(PLATFORM_ID);
  private readonly isBrowser = isPlatformBrowser(this.platformId);
  private deferredPrompt: BeforeInstallPromptEvent | null = null;

  // Se√±ales reactivas
  readonly canInstall = signal(false);
  readonly isInstalled = signal(false);
  readonly showInstallPrompt = signal(false);

  constructor() {
    if (this.isBrowser) {
      this.initializeInstallPrompt();
      this.checkIfInstalled();
    }
  }

  /**
   * Inicializa el evento de instalaci√≥n
   */
  private initializeInstallPrompt(): void {
    window.addEventListener('beforeinstallprompt', (event: Event) => {
      const beforeInstallPrompt = event as BeforeInstallPromptEvent;
      // Prevenir el prompt autom√°tico de Chrome
      beforeInstallPrompt.preventDefault();

      // Guardar el evento para usarlo despu√©s
      this.deferredPrompt = beforeInstallPrompt;
      this.canInstall.set(true);

      console.log('‚úÖ PWA instalable detectada');

      // Mostrar prompt personalizado despu√©s de 30 segundos
      setTimeout(() => {
        if (!this.isInstalled() && this.canInstall()) {
          this.showInstallPrompt.set(true);
        }
      }, 30000);
    });

    // Detectar cuando se instala
    window.addEventListener('appinstalled', () => {
      console.log('‚úÖ PWA instalada exitosamente');
      this.isInstalled.set(true);
      this.canInstall.set(false);
      this.showInstallPrompt.set(false);
      this.deferredPrompt = null;

      // Vibraci√≥n de √©xito
      if ('vibrate' in navigator) {
        navigator.vibrate([200, 100, 200]);
      }
    });
  }

  /**
   * Verifica si la app ya est√° instalada
   */
  private checkIfInstalled(): void {
    // Detecta si se abri√≥ desde PWA instalada
    if (window.matchMedia('(display-mode: standalone)').matches) {
      this.isInstalled.set(true);
      console.log('‚úÖ App ya instalada (standalone mode)');
    }

    // Detecta si es iOS instalado
    const navigatorWithStandalone = window.navigator as NavigatorWithStandalone;
    if (navigatorWithStandalone.standalone === true) {
      this.isInstalled.set(true);
      console.log('‚úÖ App ya instalada (iOS standalone)');
    }
  }

  /**
   * Muestra el prompt de instalaci√≥n
   */
  async promptInstall(): Promise<boolean> {
    if (!this.deferredPrompt) {
      console.warn('No hay prompt de instalaci√≥n disponible');
      return false;
    }

    // Mostrar el prompt nativo
    this.deferredPrompt.prompt();

    // Esperar la respuesta del usuario
    const { outcome } = await this.deferredPrompt.userChoice;

    console.log(`Usuario ${outcome === 'accepted' ? 'acept√≥' : 'rechaz√≥'} la instalaci√≥n`);

    // Limpiar el prompt usado
    this.deferredPrompt = null;
    this.canInstall.set(false);
    this.showInstallPrompt.set(false);

    return outcome === 'accepted';
  }

  /**
   * Oculta el prompt personalizado
   */
  dismissPrompt(): void {
    this.showInstallPrompt.set(false);

    if (!this.isBrowser) return;

    // No mostrar de nuevo en 7 d√≠as
    const dismissedUntil = new Date();
    dismissedUntil.setDate(dismissedUntil.getDate() + 7);
    localStorage.setItem('pwa-install-dismissed', dismissedUntil.toISOString());
  }

  /**
   * Verifica si el prompt fue rechazado recientemente
   */
  private wasRecentlyDismissed(): boolean {
    if (!this.isBrowser) return false;
    const dismissedUntil = localStorage.getItem('pwa-install-dismissed');
    if (!dismissedUntil) return false;

    const dismissDate = new Date(dismissedUntil);
    return dismissDate > new Date();
  }

  /**
   * Obtiene instrucciones de instalaci√≥n seg√∫n el navegador
   */
  getInstallInstructions(): {
    browser: string;
    instructions: string[];
    icon: string;
  } | null {
    const ua = navigator.userAgent;

    // iOS Safari
    const windowWithMsStream = window as WindowWithMSStream;
    if (/iPad|iPhone|iPod/.test(ua) && !windowWithMsStream.MSStream) {
      return {
        browser: 'Safari iOS',
        instructions: [
          'Toca el bot√≥n de compartir ‚éô',
          'Despl√°zate y selecciona "A√±adir a la pantalla de inicio"',
          'Toca "A√±adir" en la esquina superior derecha',
        ],
        icon: 'üì±',
      };
    }

    // Chrome Android
    if (/Chrome/.test(ua) && /Android/.test(ua)) {
      return {
        browser: 'Chrome Android',
        instructions: [
          'Toca el men√∫ ‚ãÆ (tres puntos)',
          'Selecciona "Instalar app" o "A√±adir a pantalla de inicio"',
          'Toca "Instalar"',
        ],
        icon: 'ü§ñ',
      };
    }

    // Edge
    if (/Edg/.test(ua)) {
      return {
        browser: 'Microsoft Edge',
        instructions: [
          'Haz clic en el √≠cono de instalaci√≥n en la barra de direcciones',
          'O ve a Men√∫ > Aplicaciones > Instalar este sitio como una aplicaci√≥n',
        ],
        icon: 'üåê',
      };
    }

    // Chrome Desktop
    if (/Chrome/.test(ua)) {
      return {
        browser: 'Chrome',
        instructions: [
          'Haz clic en el √≠cono de instalaci√≥n en la barra de direcciones',
          'O ve a Men√∫ > Instalar Autorentar',
        ],
        icon: 'üíª',
      };
    }

    return null;
  }

  /**
   * Verifica las capacidades de la plataforma
   */
  getPlatformCapabilities(): {
    notifications: boolean;
    offline: boolean;
    backgroundSync: boolean;
    fileHandling: boolean;
  } {
    return {
      notifications: 'Notification' in window && 'PushManager' in window,
      offline: 'serviceWorker' in navigator,
      backgroundSync: 'serviceWorker' in navigator && 'SyncManager' in window,
      fileHandling: 'launchQueue' in window,
    };
  }
}
