import { Component, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';
import { CommonModule } from '@angular/common';
import { IconComponent } from '../icon/icon.component';

@Component({
  selector: 'app-availability-alert',
  standalone: true,
  imports: [CommonModule, IconComponent],
  templateUrl: './availability-alert.component.html',
  styleUrls: ['./availability-alert.component.css'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class AvailabilityAlertComponent {
  @Input() count = 0;
  @Input() zone = 'tu zona';
  @Output() expandRadius = new EventEmitter<void>();
}

<div
  *ngIf="count > 0 && count <= 3"
  class="availability-alert bg-warning-bg dark:bg-warning-900/20 border border-warning-border dark:border-warning-800 rounded-lg p-3 mb-4"
>
  <div class="flex items-center justify-between">
    <div class="flex items-center gap-2">
      <app-icon
        name="alert-circle"
        [size]="20"
        cssClass="text-warning-text dark:text-warning-400"
      />
      <p class="text-sm font-medium text-warning-strong dark:text-warning-200">
        Solo quedan <strong>{{ count }}</strong> {{ count === 1 ? 'auto' : 'autos' }} en {{ zone }}
      </p>
    </div>
    <button
      type="button"
      (click)="expandRadius.emit()"
      class="text-xs font-semibold text-warning-strong dark:text-warning-300 hover:text-warning-strong dark:hover:text-warning-100 transition-colors"
    >
      Ampliar radio →
    </button>
  </div>
</div>
